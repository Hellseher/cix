    #!/usr/bin/env bash

    FILE_TO_SEND="send_me.there"
    HOST_NAME="waiting.for.me"
    USER_NAME="are_you"

    while true; do
        rsync -avz --append "$FILE_TO_SEND" "$USER_NAME"@"$HOST_NAME"
        [[ "$?" = "0" ]] && break
        echo "File is not tranfared, resume."
    done
