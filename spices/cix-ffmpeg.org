- ffmpeg-161116232611 :: Simple screen cast.
#+BEGIN_SRC sh
  #!/usr/bin/env bash

  OUTPUT_PATH="$HOME/screencasts"
  [[ -d "$OUTPUT_PATH" ]] ||  mkdir -p "$OUTPUT_PATH"

  ffmpeg -f x11grab \
         -r 25 \
         -s 800x600 \
         -i :0.0 \
         "$OUTPUT_PATH"/output_file.mpg
#+END_SRC
